    public FilterTypeTemp updateFilter(Collection<> values) {
        Map< , CheckState> oldMap = new HashMap<>();
        for (FilterItem item : this.items) {
            Item item1 = (Item) item;
            oldMap.put(item1.getSequence(), item.isChecked());
        }
        this.clear();
        for (  value : values) {
            Item item = new Item(value);
            CheckState checkState = oldMap.get(value);
            if (checkState != null) {
                item.setChecked(checkState);
            }
            this.addItem(item);
        }
        return this;
    }
